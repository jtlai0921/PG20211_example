public class MyMain {
    public static void main(String[] args) {
        int a[];       	//宣告放置陣列用的變數
        a = new int [3];  	//確保陣列用的區域，指派到變中

        a[0]=0;
        a[1]=1;
        a[2]=2;

        System.out.println(a[0]);
        System.out.println(a[1]);
        System.out.println(a[2]);
    }
}
