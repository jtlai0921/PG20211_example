

#include <iostream>
using namespace std;

class ClassA{
  ClassA* next;
  int low;
  int high;
public:
  ClassA(int l,int h){
    low=l;
    high=h;
    next=0;
  }
  void setNext(ClassA* n){
    next=n;
  }
  void process(int val){
    if(low<=val && val<high){@	//Ž©•ª‚ÌÓ”C”ÍˆÍ‚È‚çˆ—
      cout << low << " <= " << val << " < "									<< high << endl;
    }
    else if(next!=0){@		//ŽŸ‚ª‚ ‚ê‚Î‚½‚ç‚¢‰ñ‚µ‚É‚·‚é
      next->process(val);
    }
  }
};


int main(void)
{
  ClassA a1(0,30);
  ClassA a2(30,70);
  ClassA a3(70,100);
  a1.setNext(&a2);
  a2.setNext(&a3);

  a1.process(20);
  a1.process(50);
  a1.process(90);
}
