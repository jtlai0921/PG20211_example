

class MyMain{
    public static void main(String[] args){
	Element[] elements=new Element[]{
		new ElementB("Hello"),
		new ElementA(42),
		new ElementB("peoples")};
	VisitorPrint visit=new VisitorPrint();
	for(int i=0;i<elements.length;i++){
	    elements[i].Accept(visit);
	}
    }
}

interface Element{
    void Accept(Visitor v);
}

interface Visitor{
    void VisitA(ElementA e);　	//ElementAについての処理
    void VisitB(ElementB e);　	//ElementBについての処理
}

class ElementA implements Element{
    int n;
    public ElementA(int i){
	n=i;
    }

    public void Accept(Visitor v){
	v.VisitA(this);
    }
}

class ElementB implements Element{
    String msg;
    public ElementB(String s){
	msg=s;
    }
    public void Accept(Visitor v){
	v.VisitB(this);
    }
}

class VisitorPrint implements Visitor{　	//処理の例：印刷
    public void VisitA(ElementA e){
	System.out.print(e.n + " ");
    }
    public void VisitB(ElementB e){
	System.out.print(e.msg + " ");
    }
}
